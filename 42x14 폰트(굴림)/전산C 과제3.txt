/* 과제 2

	}

	if (ch == '-')
	{
		value = (float)a - b;
	}

	if (ch == '*')
	{
		value = (float)a * b;
	}

	if (ch == '/')
	{
		value = (float)a / b;
	}

	return value;
}

int main(void)
{

	float result = 0.0;
